const SHEET_ID = "1txSmMBcdAgpXRz5el-scefkGrzzKrn8lbS1BqF1X5hk"; const S_ORDINI="Ordini",S_RIM="Rimanenze",S_PRES="Presenze",S_COP="Coperti"; function getOrCreateSheet(ss,name,headers,color){ var s=ss.getSheetByName(name); if(!s){s=ss.insertSheet(name);s.appendRow(headers); s.getRange(1,1,1,headers.length).setFontWeight("bold").setBackground(color||"#1e293b").setFontColor("#fff"); s.setFrozenRows(1);} return s; } function doPost(e){ try{ var data=JSON.parse(e.postData.contents); var ss=SpreadsheetApp.openById(SHEET_ID); var tipo=data.tipo||"ordini"; if(tipo==="ordini"){ var sh=getOrCreateSheet(ss,S_ORDINI,["ID","Data","Ora","Fornitore","Azienda","Giorno Consegna","Prodotto","Quantita","Note","WA"],"#1e293b"); (data.orders||[]).forEach(function(o){o.items.forEach(function(it){sh.appendRow([o.id,o.date,o.time,o.supplier,o.company||"",o.deliveryDay,it.name,it.qty,o.note||"",o.waSent?"SI":"NO"]);});}); return ok({added:(data.orders||[]).length}); } if(tipo==="rimanenze"){ var sh=getOrCreateSheet(ss,S_RIM,["Data","Coperti","Arrosticini","Pizze","Hamburger","Tag.Rosse","Tag.Bianche","Turno","Note"],"#1e3a5f"); (data.rimanenze||[]).forEach(function(r){sh.appendRow([r.data,r.coperti||0,(r.articoli&&r.articoli.arrosticini)||0,(r.articoli&&r.articoli.pizze)||0,(r.articoli&&r.articoli.hamburger)||0,(r.articoli&&r.articoli.tagliateR)||0,(r.articoli&&r.articoli.tagliateB)||0,r.turno||"",r.note||""]);}); return ok({added:(data.rimanenze||[]).length}); } if(tipo==="presenze"){ var sh=getOrCreateSheet(ss,S_PRES,["Dipendente","Ruolo","Anno","Mese","Data","Stato"],"#3b1f5f"); (data.presenze||[]).forEach(function(r){sh.appendRow([r.nome,r.ruolo||"",r.anno,r.mese,r.data,r.stato]);}); return ok({added:(data.presenze||[]).length}); } if(tipo==="coperti"){ var sh=getOrCreateSheet(ss,S_COP,["Data","Giorno","Coperti","Arrosticini","Pizze","Hamburger","Tag.Rosse","Tag.Bianche"],"#1f5f3b"); var gg=["Domenica","Lunedi","Martedi","Mercoledi","Giovedi","Venerdi","Sabato"]; (data.storico||[]).forEach(function(s){sh.appendRow([s.data,gg[new Date(s.data).getDay()]||"",s.coperti||0,s.arrosticini||0,s.pizze||0,s.hamburger||0,s.tagliateR||0,s.tagliateB||0]);}); return ok({added:(data.storico||[]).length}); } return err("Tipo non riconosciuto: "+tipo); }catch(ex){return err(ex.message);} } function doGet(){return ok({status:"online",version:"2.0"});} function ok(d){d.ok=true;return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON);} function err(m){return ContentService.createTextOutput(JSON.stringify({ok:false,error:m})).setMimeType(ContentService.MimeType.JSON);}
